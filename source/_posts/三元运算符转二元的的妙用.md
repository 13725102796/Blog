# 三元运算符
var obj = {a: 1}

var a = obj.a ? obj.a : undefined

# ?.的妙用
var obj = {a: 1}

var a = obj?.a 

# 在jsx的语法中，经常需要模版语法根据条件渲染，此时可以缩减写法
const obj = {arr: [1,2,3], }
obj?.arr.map((item)=>{
  return (
    <div>{item}</div>
  )
})
# 因为obj.arr1不存在所以不会执行
obj.arr1?.arr1.map((item)=>{
  return (
    <div>{item}</div>
  )
})
